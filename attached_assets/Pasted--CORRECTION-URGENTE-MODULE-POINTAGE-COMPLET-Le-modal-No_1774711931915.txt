Utiliser dans React:
  import SignaturePad from 'signature_pad';

  useEffect(() => {
    const canvas = document.getElementById(`sig-arrival-${workerId}`);
    const pad = new SignaturePad(canvas, {
      backgroundColor: 'rgb(255,255,255)',
      penColor: 'rgb(0,0,0)',
      minWidth: 1,
      maxWidth: 2.5
    });
    // Stocker pad dans ref: sigPads.current[workerId] = pad

  // Effacer:
  sigPads.current[workerId].clear();

  // Récupérer image:
  const imgBase64 = sigPads.current[workerId].toDataURL('image/png');

  // Vérifier si signé:
  const isSigned = !sigPads.current[workerId].isEmpty();

## CORRECTIONS BASE DE DONNÉES

Vérifier et créer si manquant:

CREATE TABLE IF NOT EXISTS pointage_sheets (
  id               SERIAL PRIMARY KEY,
  project_id       INTEGER REFERENCES projects(id) NOT NULL,
  date             DATE NOT NULL,
  created_by       INTEGER REFERENCES users(id) NOT NULL,
  chef_signature   TEXT,
  chef_signed_at   TIMESTAMP,
  status           VARCHAR(20) DEFAULT 'BROUILLON',
  total_amount_due INTEGER DEFAULT 0,
  admin_comment    TEXT,
  locked           BOOLEAN DEFAULT false,
  created_at       TIMESTAMP DEFAULT NOW(),
  UNIQUE(project_id, date)
);

CREATE TABLE IF NOT EXISTS pointage_entries (
  id                   SERIAL PRIMARY KEY,
  sheet_id             INTEGER REFERENCES pointage_sheets(id) ON DELETE CASCADE,
  worker_id            INTEGER NOT NULL,
  status               VARCHAR(20) DEFAULT 'PRESENT',
  arrival_time         TIME,
  arrival_signature    TEXT,
  arrival_signed_at    TIMESTAMP,
  departure_time       TIME,
  departure_signature  TEXT,
  departure_signed_at  TIMESTAMP,
  hours_worked         DECIMAL(4,2),
  overtime_hours       DECIMAL(3,2) DEFAULT 0,
  pay_mode             VARCHAR(10) DEFAULT 'PAR_JOUR',
  daily_rate           INTEGER,
  task_id              INTEGER REFERENCES tasks(id),
  task_amount          INTEGER,
  task_progress_pct    INTEGER DEFAULT 100,
  amount_due           INTEGER DEFAULT 0,
  notes                TEXT,
  created_at           TIMESTAMP DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS project_workers (
  project_id INTEGER REFERENCES projects(id) ON DELETE CASCADE,
  worker_id  INTEGER REFERENCES users(id) ON DELETE CASCADE,
  assigned_at TIMESTAMP DEFAULT NOW(),
  assigned_by INTEGER REFERENCES users(id),
  PRIMARY KEY (project_id, worker_id)
);

-- Lier le personnel existant aux projets:
INSERT INTO project_workers (project_id, worker_id, assigned_by)
SELECT p.id, u.id, 1
FROM projects p, users u
WHERE u.role = 'OUVRIER'
ON CONFLICT DO NOTHING;
